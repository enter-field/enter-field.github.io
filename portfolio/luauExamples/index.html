Lanyard System | Integration for Whitehill Group's DWProx Net2+ Door System


local Players = game:GetService("Players")

local function connectLanyard(lanyard)
	if not lanyard:GetAttribute("IsLanyardBadge") then return end

	print("connecting lanyard: " .. lanyard.Name)

	for _, descendent in ipairs(lanyard:GetDescendants()) do
		if descendent:IsA("BasePart") then
			descendent.Touched:Connect(function(hit)

				if hit and hit:GetAttribute("IsNet2Reader") == true then

					local doorModel = hit
					for i = 1, 4 do
						if doorModel.Parent then
							doorModel = doorModel.Parent
						end
					end

					local api = doorModel:WaitForChild("DWPROXAPI", 5)
					if api then
						
						local RequiredClearance = hit:GetAttribute("RequiredClearance") 
						local LanyardClearance = lanyard:GetAttribute("Clearance")
						
						
						if api:IsA("BindableFunction") then
							
							if LanyardClearance >= RequiredClearance then
								api:Invoke("OPEN")
							else
								
							end
						else
							print("the API exists, but isn't a BindableFunction")
						end
					else
						print("didn't find the API at all under that door model (was it a door you touched?)")
					end
				end
			end)
		end
	end
end

local function onCharacterAdded(char)
	print("character added, name:", char.Name)
	
	for _, child in ipairs(char:GetChildren()) do
		if child:IsA("Model") and child:GetAttribute("IsLanyardBadge") == true then
			connectLanyard(child)
		end
	end
	char.DescendantAdded:Connect(function(desc)
		if desc:IsA("Model") and desc:GetAttribute("IsLanyardBadge") then
			connectLanyard(desc)
		end
	end)
end


Players.PlayerAdded:Connect(function(player)
	player.CharacterAdded:Connect(onCharacterAdded)
end)
